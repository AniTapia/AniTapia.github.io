Total Semesters

This java program determines the amount of semesters needed to take all of the computer science courses at West Chester University

This program reads in a file that states the computer science courses and their prerequisites. Based on these prerequisites, the program displays to the user what courses will be taken in each semester. There is no limit as to how many courses the user can take in each semester.

Edge.java

									
public class Edge{
	private String p;
	private String v;
	public Edge(String a, String b){
		p = a;
		v = b;
	}
	
	public String getDest(){ return v;}
	public String getSource() { return p;}
	}

Node.java

									
import java.util.ArrayList;

public class Node{
private String source;
private ArrayList< Edge> prerequisites;
public Node(String a){
source = a;
prerequisites = new ArrayList< Edge>();
}

public void addEdge(Edge x){
	prerequisites.add(x);}
public String getSource(){return source;}
public int getSize(){return prerequisites.size();}
public Edge getPrerequisite(int i){return prerequisites.get(i);}
public String toString(){
	String pre = "";
	if(prerequisites.size() == 0)return "CSC " + source + " Prerequisites: null\n";
	Edge tmp;
	for(int i = 0; i < prerequisites.size(); i++)
	{
		tmp = prerequisites.get(i);
		pre += " CSC " + tmp.getSource();
	}
	return "CSC " + source + " Prerequisites:" + pre + "\n";
}
public int contains(String s){
	Edge tmp;
	for(int i = 0; i< prerequisites.size(); i++){
		tmp = prerequisites.get(i);
		if(tmp.getSource().equals(s)) return i;
	}
	return -1;
}
public void removeEdge(int i){
	prerequisites.remove(i);
}
}

Graph.java

									
import java.util.Scanner;
import java.util.ArrayList;
import java.io.*;

import java.util.NoSuchElementException;


public class Graph{
private ArrayList< Node> courses = new ArrayList< Node>();

public Graph(String filename){
File inputFile = new File (filename);
try {   
	Scanner input = new Scanner(inputFile);

	while(input.hasNext()){
		String line = input.nextLine();
		String [] info = line.split(" ");
		Node tmp = new Node(info[0]);
		for(int i = 1; i< info.length; i++){
			Edge e = new Edge(info[i], info[0]);
			tmp.addEdge(e);
		}
		courses.add(tmp);
	}

	input.close();
}
catch (FileNotFoundException e)
{
	System.out.println("file reading fails.");
}
}


public String toString(){
String ret ="";
for(int i = 0; i< courses.size(); i++){
Node tmp = courses.get(i);
ret += tmp.toString();
}
return ret;
}

public ArrayList< Node> getNoPreRecCourses() {//return a list of courses with no prerequisites
ArrayList< Node> tmp_coursers = new ArrayList< Node>();
for(int i = 0; i < courses.size(); i++)
{
	Node tmp = courses.get(i);
	if(tmp.getSize() == 0)tmp_coursers.add(tmp);
}
return tmp_coursers;
}
public ArrayList< Node> getOutgoingCourses(Node n)//input a node, and return its outgoing courses
{
ArrayList< Node> outgoing_Courses = new ArrayList< Node>();
for(int i = 0; i < courses.size(); i++)
{
	Node tmpN = courses.get(i);
	for(int j = 0; j < tmpN.getSize(); j++)
	{
		Edge tmpE = tmpN.getPrerequisite(j);
		if(tmpE.getSource().equals(n.getSource()))outgoing_Courses.add(tmpN);
	}
}
return outgoing_Courses;
}

public void total_Semesters()
{
int total = 0;
int semester = 1;
ArrayList< Node> noIncoming = new ArrayList< Node>();
ArrayList< Node> outgoing = new ArrayList< Node>();
Node tmpN = null;
Node tmpN2 = null;
while(!courses.isEmpty())//loop until there are no more courses remaining
{
	System.out.print("\nSemester " + semester + ":");
	noIncoming = getNoPreRecCourses();//list of courses with no prerequisites
	for(int i = 0; i < noIncoming.size(); i++)//for each course with no prerequisites, remove the course and its outgoing edges
	{
		tmpN = noIncoming.get(i);
		System.out.print(" CSC " + tmpN.getSource());
		outgoing = getOutgoingCourses(tmpN);//list of outgoing courses for a node
		for(int j = 0; j < outgoing.size(); j++)//loop through each outgoing course
		{
			tmpN2 = outgoing.get(j);
			tmpN2.removeEdge(tmpN2.contains(tmpN.getSource()));//remove the course as a prerequisite
		}
		courses.remove(tmpN);//remove the course
	}
	total++;
	semester++;
}
System.out.println("\nTotal semesters: " + total);
}

public static void main(String args[]){
Graph g = new Graph("src/Courses.txt");
System.out.println(g);
g.total_Semesters();
}
