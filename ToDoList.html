To Do List

A To Do List Web App made with React

This web app is a to do list for the user. It has two sections: To Do List and Completed task. In the to do list section, the user can see the items they have to do, add new items, and check of items that are Completed. In the complted task section, the user can see which tasks they have done. They can also put a task back in the to do section.

App.js

									
import React, { Component } from 'react';
import { TodoBanner } from "./TodoBanner";
import { TodoCreator } from "./TodoCreator";
import { TodoRow } from "./TodoRow";
import { VisibilityControl } from "./VisibilityControl";

export default class App extends Component {
	constructor(props){
	super(props);
	this.state = {
		userName: "Ani",
		todoItems: [{action: "Get groceries at ShopRite", done: false},
                    {action: "Complete Deal Or No Deal Program", done: false},
                    {action: "Fix the microwave", done: false},
                    {action: "Take out the trash", done: true}],
		showCompleted: true
		}
	}
	updateNewTextValue = (event) => {
		this.setState({newItemText: event.target.value});
	}
	createNewTodo = (task) => {
		if (!this.state.todoItems.find(item => item.action === task)) {
			this.setState({
				todoItems: [...this.state.todoItems, { action: task, done: false }]
			});
		}
	}

	toggleTodo = (todo) => this.setState({ todoItems:
		this.state.todoItems.map(item => item.action === todo.action
			? { ...item, done: !item.done } : item) });
	
	todoTableRows = (doneValue) => this.state.todoItems
		.filter(item => item.done === doneValue).map(item =>
			<.TodoRow key={ item.action } item={ item }
					callback={ this.toggleTodo } />)
	
	render = () =>
		< div>
			< TodoBanner name={ this.state.userName }
				tasks={this.state.todoItems } />
			< div className="container-fluid">
				< TodoCreator callback={ this.createNewTodo } />
				< table className="table table-striped table-bordered">
					< thead>
						< tr>< th>Description< /th>< th>Done< /th>< /tr>
					< /thead>
					< tbody>{ this.todoTableRows(false) }< /tbody>
				< /table>
				< div className="bg-secondary text-white text-center p-2">
					< VisibilityControl description="Completed Tasks"
						isChecked={this.state.showCompleted}
						callback={ (checked) =>
							this.setState({ showCompleted: checked })} />
				< /div>
				
				{ this.state.showCompleted &&
					< table className="table table-striped table-bordered">
						< thead>
							< tr>< th>Description< /th>< th>Done< /th>< /tr>
						< /thead>
						< tbody>{ this.todoTableRows(true) }< /tbody>
					< /table>
				}
			< /div>
		< /div>
}

TodoBanner.js

									
import React, { Component } from 'react';

export class TodoBanner extends Component {
	render = () =>
		< h4 className="bg-primary text-white text-center p-3 m-2">
			{ this.props.name }'s To Do List
			({ this.props.tasks.filter(t => !t.done).length } items to do)
		< /h4>
}

TodoCreator.js

									
import React, { Component } from 'react';

export class TodoCreator extends Component {
	constructor(props) {
	super(props);
	this.state = { newItemText: "" }
	}
	updateNewTextValue = (event) => {
	this.setState({ newItemText: event.target.value});
	}

	createNewTodo = () => {
	this.props.callback(this.state.newItemText);
	this.setState({ newItemText: ""});
	}

	render = () =>
	< div className="my-1">
		< input className="form-control" value={ this.state.newItemText }
			onChange={ this.updateNewTextValue } />
		< button className="btn btn-primary mt-1"
			onClick={ this.createNewTodo }>Add< /button>
	< /div>
}

TodoRow.js

									
import React, { Component } from 'react';

export class TodoRow extends Component {
	render =() =>
		< tr>
			< td>{this.props.item.action}< /td>
			< td>
				< input type="checkbox" checked={this.props.item.done}
				onChange={() => this.props.callback(this.props.item)}
				/> 
			< /td>
		< /tr>
}

VisibilityControl.js

									
import React, { Component } from 'react';

export class VisibilityControl extends Component {

	render = () =>
		< div className="form-check">
			< input className="form-check-input" type="checkbox"
				checked={ this.props.isChecked }
				onChange={ (e) => this.props.callback(e.target.checked) } />
			< label className="form-check-label">
				Show { this.props.description }
			< /label>
		< /div>
}
