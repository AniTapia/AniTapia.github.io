Eight Ball

A java program that simulates the magic 8 ball

Just like the real life magic 8 ball, the user can ask a yes or no question, and then shake the magic 8 ball to recieve a response.

EightBall.java

									
import java.util.Scanner;
public class EightBall
{
	public static void main(String args[])
	{
		Scanner in = new Scanner(System.in);
		String input;
		int num;
		
		System.out.print("Enter \"0\" to quit, or ask a question: ");
		input = in.nextLine();
		
		while(!input.startsWith("0"))
		{
			System.out.println();
			num = (int)(Math.random() * 20 + 1);
			switch(num)
			{
				case 1: System.out.println("It is certain."); break;
				case 2: System.out.println("You canâ€™t handle the truth."); break;
				case 3: System.out.println("Don't count on it"); break;
				case 4: System.out.println("Concentrate and ask again."); break;
				case 5: System.out.println("Signs point to yes."); break;
				case 6: System.out.println("It is decidedly so."); break;
				case 7: System.out.println("Ask again later."); break;
				case 8: System.out.println("My reply is no."); break;
				case 9: System.out.println("As I see it, yes"); break;
				case 10: System.out.println("Most likely."); break;
				case 11: System.out.println("Cannot predict now."); break;
				case 12: System.out.println("Without a doubt."); break;
				case 13: System.out.println("Absolutely"); break;
				case 14: System.out.println("Outlook not so good."); break;
				case 15: System.out.println("Outlook good."); break;
				case 16: System.out.println("Better not tell you now."); break;
				case 17: System.out.println("Yes - definitely."); break;
				case 18: System.out.println("Very doubtful."); break;
				case 19: System.out.println("My sources say no."); break;
				case 20: System.out.println("Yes"); break;
			}
			System.out.print("\nEnter \"0\" to quit, or ask a question: ");
			input = in.nextLine();
		}
		System.out.println("\nHave a wonderful day!");
	}
}
